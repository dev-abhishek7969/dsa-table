/** * Input: nums = [1,12,-5,-6,50,3], k = 4 Output: 12.75 * (12 - 5 - 6 + 50) / 4 = 51 / 4 = 12.75 */ public double findMaxAverage(int[] nums, int k) { int windowSum = 0; for (int i = 0; i < k; i++) { windowSum = windowSum + nums[i]; } int maxSum = windowSum; int startIndex = 0; int endIndex = k; while (endIndex < nums.length) { windowSum = windowSum - nums[startIndex]; startIndex++; windowSum = windowSum + nums[endIndex]; endIndex++; maxSum = Math.max(maxSum, windowSum); } return (double) maxSum / k; }
